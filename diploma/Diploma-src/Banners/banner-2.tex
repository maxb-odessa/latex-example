\Large\bf

$l = (d-x)X + yY + zZ$
\\

$M_x = x + lx - d$
\\

$M^2 = (x-d)^2 + y^2 + z^2 -l^2$ 
\\

$Q = \sqrt{X^2 + c^2e^2M^2_x + 2cM_x - c^2M^2(1 - e^2X^2)}$ 
\\

$L = l + \frac{cM^2-c^2e^2M^2_x-2M_x}{X(ce^2M_x+1)+Q}$
\\

$x_i = x - d + LX$
\\

$y_i = y + LY$
\\

$z_i = z + LZ$
\\

$D = \sqrt{1+c^2e^2(y_i^2+z_i^2)}$
\\

$\cos{i} = \frac{Q}{D}$
\\

$\cos{i'} = \sqrt{1-\left(\frac{n}{n'}\right)^2\left(1-\cos^2{i}\right)}$
\\

$g = \cos{i'} - \frac{n}{n'}\cos{i}$
\\

$X_i = \frac{n}{n'}X - g\frac{c(1-e^2)x_i-1}{D}$
\\

$Y_i = \frac{n}{n'}Y - g\frac{cy_i}{D}$
\\

$Z_i = \frac{n}{n'}Z - g\frac{cz_i}{D}$
\\
